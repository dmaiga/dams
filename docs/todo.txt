from datetime import date
from django.db import transaction
from core.models import Agent, Recouvrement

# -----------------------------
# PARAMÈTRES
# -----------------------------
DATE_EFFET = date(2026, 1, 1)

# -----------------------------
# SUPERVISEURS (par username)
# -----------------------------
ancien_superviseur = Agent.objects.get(
    user__username="abdoulaye.kone",
    type_agent="entrepot"
)

nouveau_superviseur = Agent.objects.get(
    user__username="sidibe.mankoulako",
    type_agent="entrepot"
)

# -----------------------------
# AGENTS TERRAIN CONCERNÉS
# (ceux actuellement sous le nouveau superviseur)
# -----------------------------
agents_terrain = Agent.objects.filter(
    type_agent="terrain",
    superviseur=nouveau_superviseur,
    est_actif=True
)

print("Agents terrain concernés :", agents_terrain.count())

# -----------------------------
# REC0UVREMENTS À TRANSFÉRER
# -----------------------------
recouvrements_a_transferer = Recouvrement.objects.filter(
    superviseur=ancien_superviseur,
    agent__in=agents_terrain,
    date_recouvrement__date__gte=DATE_EFFET
)

print("Recouvrements trouvés :", recouvrements_a_transferer.count())

# 🔎 Sécurité : aperçu avant transfert
for r in recouvrements_a_transferer[:10]:
    print(
        r.id,
        r.agent_id,
        r.montant_recouvre,
        r.date_recouvrement
    )

# -----------------------------
# TRANSFERT ATOMIQUE
# -----------------------------
with transaction.atomic():
    recouvrements_a_transferer.update(
        superviseur=nouveau_superviseur
    )

print("✅ Transfert terminé avec succès")
